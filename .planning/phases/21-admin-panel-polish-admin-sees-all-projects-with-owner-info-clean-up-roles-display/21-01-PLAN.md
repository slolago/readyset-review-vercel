**1. Create `src/app/api/admin/projects/route.ts`:**

Follow the exact pattern from `src/app/api/admin/users/route.ts`. Single GET handler:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { requireAdmin } from '@/lib/auth-helpers';
import { getAdminDb } from '@/lib/firebase-admin';
```

- Call `requireAdmin(request)` — return 403 if null
- `db.collection('projects').orderBy('createdAt', 'desc').get()` — fetch all projects
- Collect unique `ownerId` values from all projects via `new Set()`
- Batch-fetch owners using `db.getAll(...ownerIds.map(id => db.collection('users').doc(id)))` — single Firestore RPC, NOT individual gets in a loop
- Build `ownerMap: Map<string, { name: string, email: string }>` from the results (filter `.exists`)
- Return enriched array:
  ```typescript
  {
    id: string,
    name: string,
    description: string,
    color: string,
    createdAt: Timestamp,
    ownerId: string,
    ownerName: string,   // from ownerMap, fallback 'Unknown'
    ownerEmail: string,  // from ownerMap, fallback ''
    collaboratorCount: number  // (project.collaborators || []).length
  }
  ```
- Wrap in try/catch, return 500 on error
- Return `{ projects: enriched }`

const fetchProjects = useCallback(async () => {
  setProjectsLoading(true);
  try {
    const token = await getIdToken();
    const res = await fetch('/api/admin/projects', {
      headers: { Authorization: `Bearer ${token}` },
    });
    if (res.ok) {
      const data = await res.json();
      setProjects(data.projects);
    }
  } catch {
    toast.error('Failed to load projects');
  } finally {
    setProjectsLoading(false);
  }
}, [getIdToken]);
