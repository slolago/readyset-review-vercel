2. Create a handler for dropping onto a folder:
   ```typescript
   const handleFolderDrop = useCallback(async (targetFolderId: string, e: React.DragEvent) => {
     e.preventDefault();
     e.stopPropagation();
     setDragOverFolderId(null);

     const raw = e.dataTransfer.getData('application/x-frame-move');
     if (!raw) return;

     let payload: { ids: string[] };
     try { payload = JSON.parse(raw); } catch { return; }

     const ids = payload.ids;
     if (!ids?.length) return;

     // Cannot move a folder into itself
     if (ids.includes(targetFolderId)) {
       toast.error("Cannot move a folder into itself");
       return;
     }

     // Temporarily set selectedIds to the dragged ids so handleMoveSelected picks them up
     setSelectedIds(new Set(ids));
     // Use a microtask to ensure state is set before calling move
     // Actually, handleMoveSelected reads from selectedIds state which won't be
     // updated yet in the same tick. Instead, inline the move logic directly:
     try {
       const token = await getIdToken();
       const assetIds = ids.filter((id) => assets.some((a) => a.id === id));
       const folderIds = ids.filter((id) => folders.some((f) => f.id === id));

       await Promise.all([
         ...assetIds.map((id) =>
           fetch(`/api/assets/${id}`, {
             method: 'PUT',
             headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
             body: JSON.stringify({ folderId: targetFolderId }),
           })
         ),
         ...folderIds.map((id) =>
           fetch(`/api/folders/${id}`, {
             method: 'PUT',
             headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
             body: JSON.stringify({ parentId: targetFolderId }),
           })
         ),
       ]);

       toast.success(`Moved ${ids.length} item(s)`);
       setSelectedIds(new Set());
       refetchAssets();
       fetchFolders();
     } catch {
       toast.error('Failed to move items');
     }
   }, [assets, folders, getIdToken, refetchAssets, fetchFolders]);
   ```

   Note: We inline the move API calls rather than calling `handleMoveSelected` because that function reads `selectedIds` from React state, which won't reflect the dragged IDs synchronously. Inlining avoids a race condition.
