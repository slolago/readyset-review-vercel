```tsx
'use client';

import type { Asset } from '@/types';
import { formatBytes, formatDuration } from '@/lib/utils';

interface FileInfoPanelProps {
  asset: Asset;
}

function gcd(a: number, b: number): number {
  return b === 0 ? a : gcd(b, a % b);
}

function aspectRatioLabel(w: number, h: number): string {
  const d = gcd(w, h);
  return `${w / d}:${h / d}`;
}

interface InfoRowProps {
  label: string;
  value: string | number | undefined | null;
}

function InfoRow({ label, value }: InfoRowProps) {
  const display = value === undefined || value === null || value === '' ? '—' : String(value);
  return (
    <div className="flex flex-col gap-0.5 py-2.5 border-b border-frame-border last:border-0">
      <span className="text-xs text-frame-textMuted uppercase tracking-wide">{label}</span>
      <span className="text-sm text-white break-all">{display}</span>
    </div>
  );
}

export function FileInfoPanel({ asset }: FileInfoPanelProps) {
  const hasResolution = asset.width !== undefined && asset.height !== undefined;
  const aspectRatio =
    hasResolution ? aspectRatioLabel(asset.width!, asset.height!) : undefined;

  const uploadDate = asset.createdAt
    ? asset.createdAt.toDate().toLocaleDateString(undefined, {
        year: 'numeric',
        month: 'short',
        day: 'numeric',
      })
    : undefined;

  return (
    <div className="flex-1 overflow-y-auto px-4 py-2">
      <InfoRow label="Filename" value={asset.name} />
      <InfoRow label="Type" value={asset.mimeType} />
      <InfoRow label="Size" value={formatBytes(asset.size)} />
      {asset.type === 'video' && (
        <InfoRow
          label="Duration"
          value={asset.duration !== undefined ? formatDuration(asset.duration) : undefined}
        />
      )}
      <InfoRow
        label="Resolution"
        value={hasResolution ? `${asset.width} × ${asset.height}` : undefined}
      />
      <InfoRow label="Aspect Ratio" value={aspectRatio} />
      <InfoRow label="FPS" value={asset.frameRate} />
      <InfoRow label="Uploaded By" value={asset.uploadedBy} />
      <InfoRow label="Date" value={uploadDate} />
      <InfoRow label="Version" value={asset.version} />
    </div>
  );
}
```
