```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getAuthenticatedUser } from '@/lib/auth-helpers';
import { getAdminDb } from '@/lib/firebase-admin';

export async function POST(request: NextRequest) {
  const user = await getAuthenticatedUser(request);
  if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  try {
    const { sourceId, targetId } = await request.json();
    if (!sourceId || !targetId) {
      return NextResponse.json({ error: 'sourceId and targetId required' }, { status: 400 });
    }

    // Guard: self-merge (P28-05)
    if (sourceId === targetId) {
      return NextResponse.json({ error: 'Cannot merge asset with itself' }, { status: 400 });
    }

    const db = getAdminDb();

    // Fetch both root docs in parallel
    const [sourceDoc, targetDoc] = await Promise.all([
      db.collection('assets').doc(sourceId).get(),
      db.collection('assets').doc(targetId).get(),
    ]);

    if (!sourceDoc.exists) return NextResponse.json({ error: 'Source not found' }, { status: 404 });
    if (!targetDoc.exists) return NextResponse.json({ error: 'Target not found' }, { status: 404 });

    const sourceData = sourceDoc.data() as any;
    const targetData = targetDoc.data() as any;

    // Derive group IDs — legacy assets may lack versionGroupId field (P28-02)
    const sourceGroupId = sourceData.versionGroupId || sourceId;
    const targetGroupId = targetData.versionGroupId || targetId;

    // Guard: same-group merge (P28-06)
    if (sourceGroupId === targetGroupId) {
      return NextResponse.json({ error: 'Assets are already in the same version group' }, { status: 400 });
    }

    // Fetch both groups in parallel
    const [sourceGroupSnap, targetGroupSnap] = await Promise.all([
      db.collection('assets').where('versionGroupId', '==', sourceGroupId).get(),
      db.collection('assets').where('versionGroupId', '==', targetGroupId).get(),
    ]);

    // Build source group docs — include root if missing from snap (legacy assets) (P28-02)
    let sourceDocs: any[] = sourceGroupSnap.docs.map((d) => ({ id: d.id, ...d.data() }));
    if (!sourceDocs.some((d) => d.id === sourceGroupId)) {
      const rootDoc = await db.collection('assets').doc(sourceGroupId).get();
      if (rootDoc.exists) sourceDocs.push({ id: rootDoc.id, ...rootDoc.data() });
    }

    // Build target group docs — include root if missing from snap (legacy assets)
    let targetDocs: any[] = targetGroupSnap.docs.map((d) => ({ id: d.id, ...d.data() }));
    if (!targetDocs.some((d) => d.id === targetGroupId)) {
      const rootDoc = await db.collection('assets').doc(targetGroupId).get();
      if (rootDoc.exists) targetDocs.push({ id: rootDoc.id, ...rootDoc.data() });
    }

    // Find max version in target group (P28-03)
    const maxTargetVersion = Math.max(...targetDocs.map((d) => d.version || 1));

    // Sort source group ascending by version before renumbering (P28-03)
    const sortedSource = [...sourceDocs].sort((a, b) => (a.version || 1) - (b.version || 1));

    // Single atomic batch write (P28-04): update all source docs
    const batch = db.batch();
    sortedSource.forEach((doc, i) => {
      batch.update(db.collection('assets').doc(doc.id), {
        versionGroupId: targetGroupId,
        version: maxTargetVersion + 1 + i,
      });
    });
    await batch.commit();

    return NextResponse.json({ merged: sortedSource.length }, { status: 200 });
  } catch (err) {
    console.error('POST assets/merge-version error:', err);
    return NextResponse.json({ error: 'Failed to merge version groups' }, { status: 500 });
  }
}
```
