    Replacement body:
    ```ts
    const addComment = async (
      commentData: { /* same shape as before */ },
      projectId: string
    ): Promise<boolean> => {
      if (!assetId) return false;

      // Build optimistic comment. Use crypto.randomUUID() with a `temp-` prefix
      // so reconciliation can match by tempId and so real comments (server-assigned
      // IDs) never collide.
      const tempId = `temp-${typeof crypto !== 'undefined' && crypto.randomUUID ? crypto.randomUUID() : `${Date.now()}-${Math.random()}`}`;
      const nowIso = new Date().toISOString();
      const optimistic = {
        id: tempId,
        assetId,
        projectId,
        text: commentData.text,
        timestamp: commentData.timestamp,
        inPoint: commentData.inPoint,
        outPoint: commentData.outPoint,
        annotation: commentData.annotation,
        parentId: commentData.parentId ?? null,
        authorName: commentData.authorName,
        authorEmail: commentData.authorEmail,
        reviewLinkId: commentData.reviewLinkId,
        resolved: false,
        createdAt: nowIso,
      } as Comment;

      setComments((prev) => [...prev, optimistic]);

      try {
        const headers: HeadersInit = { 'Content-Type': 'application/json' };
        if (!reviewToken) {
          const token = await getIdToken();
          if (token) headers['Authorization'] = `Bearer ${token}`;
        }
        const res = await fetch('/api/comments', {
          method: 'POST',
          headers,
          body: JSON.stringify({ ...commentData, assetId, projectId }),
        });
        if (!res.ok) {
          // Rollback the temp comment.
          setComments((prev) => prev.filter((c) => c.id !== tempId));
          return false;
        }
        // Server returns the created comment. Swap the temp in place so
        // thread/reply parentId relationships survive reconciliation.
        const real = (await res.json()) as Comment;
        setComments((prev) => prev.map((c) => (c.id === tempId ? real : c)));
        return true;
      } catch {
        // Network error / JSON parse error ‚Äî same rollback.
        setComments((prev) => prev.filter((c) => c.id !== tempId));
        return false;
      }
    };
    ```

    **Reply threading note:** The `parentId` is part of `commentData` and is copied into the optimistic object, so child comments render under the correct parent immediately. After reconciliation, the server's `real` comment also carries `parentId` ‚Äî `getReplies(parent.id)` in `CommentSidebar` keeps working across the swap.
