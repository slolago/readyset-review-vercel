    ```typescript
    import type { ReactNode } from 'react';

    export type ActionTarget = 'asset' | 'folder' | 'mixed';

    /** Unified item shape — superset of ContextMenu MenuItem and Dropdown DropdownItem.
     *  Consumers map `divider` <-> `dividerBefore` when feeding into Dropdown. */
    export interface BrowserAction {
      id: string;                // stable key for dedup + testing (e.g. 'rename', 'duplicate')
      label: string;
      icon?: ReactNode;
      onClick: () => void;
      disabled?: boolean;
      danger?: boolean;
      dividerBefore?: boolean;   // used by ContextMenu
    }

    /** Handlers the consumer must supply — null/undefined means "this action is
     *  not supported in this context and will be omitted from the returned list". */
    export interface ActionContext {
      onOpen?: () => void;
      onRename?: () => void;
      onDuplicate?: () => void;
      onCopyTo?: () => void;
      onMoveTo?: () => void;
      onUploadVersion?: () => void;    // asset-only
      onStackOnto?: () => void;        // asset-only
      onManageVersions?: () => void;   // asset-only
      onDownload?: () => void;
      onGetLink?: () => void;
      onCreateReviewLink?: () => void;
      onAddToReviewLink?: () => void;
      onSetStatus?: (status: 'approved' | 'needs_revision' | 'in_review' | null) => void; // asset-only
      onDelete?: () => void;
      /** Icon factory — kept here so the action file doesn't hard-depend on lucide-react.
       *  Consumer supplies the icon map; the helper returns undefined icon if no match. */
      icons?: Partial<Record<
        | 'open' | 'rename' | 'duplicate' | 'copyTo' | 'moveTo' | 'uploadVersion'
        | 'stackOnto' | 'manageVersions' | 'download' | 'getLink'
        | 'createReviewLink' | 'addToReviewLink'
        | 'approved' | 'needsRevision' | 'inReview' | 'clearStatus' | 'delete',
        ReactNode
      >>;
    }

    export function buildFileBrowserActions(
      target: ActionTarget,
      ctx: ActionContext
    ): BrowserAction[] {
      const items: BrowserAction[] = [];
      const push = (a: BrowserAction | null) => { if (a) items.push(a); };

      // Open — asset/folder only, skipped for mixed
      if (target !== 'mixed' && ctx.onOpen)
        push({ id: 'open', label: 'Open', icon: ctx.icons?.open, onClick: ctx.onOpen });

      if (ctx.onRename && target !== 'mixed')
        push({ id: 'rename', label: 'Rename', icon: ctx.icons?.rename, onClick: ctx.onRename });

      if (ctx.onDuplicate)
        push({ id: 'duplicate', label: 'Duplicate', icon: ctx.icons?.duplicate, onClick: ctx.onDuplicate });

      if (ctx.onCopyTo)
        push({ id: 'copyTo', label: 'Copy to', icon: ctx.icons?.copyTo, onClick: ctx.onCopyTo });

      if (ctx.onMoveTo)
        push({ id: 'moveTo', label: 'Move to', icon: ctx.icons?.moveTo, onClick: ctx.onMoveTo });

      // Asset-only version operations
      if (target === 'asset') {
        if (ctx.onUploadVersion)
          push({ id: 'uploadVersion', label: 'Upload new version', icon: ctx.icons?.uploadVersion, onClick: ctx.onUploadVersion });
        if (ctx.onStackOnto)
          push({ id: 'stackOnto', label: 'Stack onto…', icon: ctx.icons?.stackOnto, onClick: ctx.onStackOnto });
        if (ctx.onManageVersions)
          push({ id: 'manageVersions', label: 'Manage version stack', icon: ctx.icons?.manageVersions, onClick: ctx.onManageVersions });
      }

      if (ctx.onDownload)
        push({ id: 'download', label: 'Download', icon: ctx.icons?.download, onClick: ctx.onDownload });

      if (ctx.onGetLink && target === 'asset')
        push({ id: 'getLink', label: 'Get link', icon: ctx.icons?.getLink, onClick: ctx.onGetLink });

      if (ctx.onCreateReviewLink)
        push({ id: 'createReviewLink', label: 'Create review link', icon: ctx.icons?.createReviewLink, onClick: ctx.onCreateReviewLink, dividerBefore: true });
      if (ctx.onAddToReviewLink)
        push({ id: 'addToReviewLink', label: 'Add to review link…', icon: ctx.icons?.addToReviewLink, onClick: ctx.onAddToReviewLink });

      // Review status — asset-only
      if (target === 'asset' && ctx.onSetStatus) {
        push({ id: 'approved', label: 'Approved', icon: ctx.icons?.approved, onClick: () => ctx.onSetStatus!('approved'), dividerBefore: true });
        push({ id: 'needsRevision', label: 'Needs Revision', icon: ctx.icons?.needsRevision, onClick: () => ctx.onSetStatus!('needs_revision') });
        push({ id: 'inReview', label: 'In Review', icon: ctx.icons?.inReview, onClick: () => ctx.onSetStatus!('in_review') });
        push({ id: 'clearStatus', label: 'Clear status', icon: ctx.icons?.clearStatus, onClick: () => ctx.onSetStatus!(null) });
      }

      if (ctx.onDelete)
        push({ id: 'delete', label: 'Delete', icon: ctx.icons?.delete, onClick: ctx.onDelete, danger: true, dividerBefore: true });

      return items;
    }
    ```

    No default export; named exports only. No imports from components — this file is pure data.

    Add defense-in-depth so the folder card's navigate-on-click handler cannot fire as a side-effect of a context menu interaction. Three specific defenses:

    **Defense 1 — guard the onClick against synthesized right-click activation**
    Currently: `if (e.button !== 0) return;`. Extend to also bail if the click target is inside a portaled menu (shouldn't happen — portal is at body — but cheap insurance):
    ```typescript
    onClick={(e) => {
      if (e.button !== 0) return;
      // Don't navigate when the user is interacting with a portaled context menu
      // that happens to land over the card (portal is at body, but this guard
      // costs nothing and documents intent).
      const target = e.target as HTMLElement;
      if (target.closest('[role="menu"]')) return;
      const url = `/projects/${projectId}/folders/${folder.id}${ancestorPath ? `?path=${ancestorPath}` : ''}`;
      router.push(url);
    }}
    ```

    **Defense 2 — swallow the synthetic click that some platforms fire after contextmenu+mouseup**
    Add a ref `const suppressNextClickRef = useRef(false);` inside `FolderCard`. In `onContextMenu` (after the existing `e.preventDefault(); e.stopPropagation(); ctxMenu.open(...)`), set `suppressNextClickRef.current = true;` then schedule `setTimeout(() => { suppressNextClickRef.current = false; }, 300);`. In `onClick` at the top, check `if (suppressNextClickRef.current) { suppressNextClickRef.current = false; return; }`. This guarantees that even if the OS/browser dispatches a click immediately after the context menu (which Linux Chromium is known to do), the card will NOT navigate.

    **Defense 3 — belt + suspenders: preventDefault the onMouseDown for right-button**
    Add:
    ```typescript
    onMouseDown={(e) => {
      if (e.button === 2) e.preventDefault();
    }}
    ```
    to the FolderCard root div. This prevents any default mousedown-initiated action (like focus + synthetic-click) the browser might otherwise do on right-click before contextmenu fires.
