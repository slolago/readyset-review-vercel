```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getAuthenticatedUser, canAccessProject } from '@/lib/auth-helpers';
import { getAdminDb } from '@/lib/firebase-admin';

export async function GET(request: NextRequest) {
  const user = await getAuthenticatedUser(request);
  if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { searchParams } = new URL(request.url);
  const projectId = searchParams.get('projectId');
  const folderId = searchParams.get('folderId') || null;

  if (!projectId) return NextResponse.json({ error: 'projectId required' }, { status: 400 });

  const hasAccess = await canAccessProject(user.id, projectId);
  if (!hasAccess) return NextResponse.json({ error: 'Forbidden' }, { status: 403 });

  const db = getAdminDb();
  // Fetch ALL assets for the project (no folderId filter at DB level — filtering in memory)
  const snap = await db.collection('assets').where('projectId', '==', projectId).get();
  const all = snap.docs.map(d => ({ id: d.id, ...d.data() } as any));
  // all assets have: folderId (string | null), size (number), projectId
}
```

Create src/app/api/assets/size/route.ts. This is a new file in a new subdirectory — Next.js will automatically treat it as a nested route at /api/assets/size.

Auth pattern: identical to src/app/api/assets/route.ts — use `getAuthenticatedUser` and `canAccessProject`.

Query parameters:
- `projectId` (required) — the project to sum
- `folderId` (optional) — if present, sum the target folder and all descendants; if absent, sum the entire project

Logic:

1. Auth + access check:
   ```typescript
   const user = await getAuthenticatedUser(request);
   if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
   const { searchParams } = new URL(request.url);
   const projectId = searchParams.get('projectId');
   const folderId = searchParams.get('folderId') || null;
   if (!projectId) return NextResponse.json({ error: 'projectId required' }, { status: 400 });
   const hasAccess = await canAccessProject(user.id, projectId);
   if (!hasAccess) return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
   ```

2. Fetch ALL assets for the project (no folderId filter — needed for recursive walk):
   ```typescript
   const db = getAdminDb();
   const snap = await db.collection('assets').where('projectId', '==', projectId).get();
   const allAssets = snap.docs.map(d => ({ id: d.id, ...d.data() } as any));
   ```

3. Fetch ALL folders for the project (needed to resolve folder descendants):
   ```typescript
   const foldersSnap = await db.collection('folders').where('projectId', '==', projectId).get();
   const allFolders = foldersSnap.docs.map(d => ({ id: d.id, ...d.data() } as any));
   ```

4. Determine which folder IDs are "in scope":
   - If folderId is null (project root requested): all assets count — set `targetFolderIds = null` to indicate "all".
   - If folderId is set: build the set of the target folder plus all descendant folder IDs using `parentId` traversal:
     ```typescript
     function getDescendantFolderIds(rootId: string, folders: any[]): Set<string> {
       const result = new Set<string>([rootId]);
       let changed = true;
       while (changed) {
         changed = false;
         for (const f of folders) {
           if (!result.has(f.id) && f.parentId && result.has(f.parentId)) {
             result.add(f.id);
             changed = true;
           }
         }
       }
       return result;
     }
     const scopedFolderIds = getDescendantFolderIds(folderId, allFolders);
     ```

5. Sum asset sizes:
   ```typescript
   let sizeBytes = 0;
   for (const asset of allAssets) {
     const inScope = folderId === null
       ? true
       : scopedFolderIds!.has(asset.folderId ?? '');
     // Also include assets directly in the target folder (folderId match)
     // and assets in root when root is requested (folderId null)
     if (inScope) {
       sizeBytes += (asset.size as number) || 0;
     }
   }
   ```

   Note on root: when folderId param is null, all project assets are included (project total). When folderId param is a folder ID, only assets in that folder or its descendants are included. Assets at the project root have `folderId === null`; a `scopedFolderIds.has(null)` check would be false, so root assets are only included when no folderId param is given — this is correct behaviour.

6. Return:
   ```typescript
   return NextResponse.json({ sizeBytes });
   ```

7. Wrap everything in try/catch — return 500 on failure.

Imports: `NextRequest`, `NextResponse` from 'next/server'; `getAuthenticatedUser`, `canAccessProject` from '@/lib/auth-helpers'; `getAdminDb` from '@/lib/firebase-admin'.
